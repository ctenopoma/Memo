履歴が不要で、ブラウザ上で完結するシンプルなルーレットアプリですね。
データの保存（データベース）が不要であるため、HTML / CSS / JavaScriptのみの静的なフロントエンドアプリを作成し、軽量なNginxコンテナで配信する構成が最もシンプルでパフォーマンスも良く、要件にぴったり合致します。
以下に、開発の手順とベースとなるコード構成をご提案します。
1. ディレクトリ構成
まずは、作業用のフォルダを作成し、以下の4つのファイルを用意します。
roulette-app/
├── Dockerfile
├── index.html
├── style.css
└── script.js

script.js (ロジックと描画)
比率に応じた角度の計算、Canvasでの円グラフ描画、そして回転のアニメーションと当選判定を行います。
let participants = [];
const colors = ['#FFB6C1', '#87CEFA', '#98FB98', '#FFD700', '#FFA07A', '#DDA0DD'];
let currentRotation = 0;

function addParticipant() {
    const name = document.getElementById('nameInput').value;
    const weight = parseInt(document.getElementById('weightInput').value);

    if (name && weight > 0) {
        participants.push({ name, weight });
        document.getElementById('nameInput').value = '';
        updateList();
        drawRoulette();
    }
}

function updateList() {
    const list = document.getElementById('participantList');
    list.innerHTML = '';
    participants.forEach(p => {
        const li = document.createElement('li');
        li.textContent = `${p.name} (比率: ${p.weight})`;
        list.appendChild(li);
    });
}

function drawRoulette() {
    const canvas = document.getElementById('rouletteCanvas');
    const ctx = canvas.getContext('2d');
    const totalWeight = participants.reduce((sum, p) => sum + p.weight, 0);
    
    ctx.clearRect(0, 0, canvas.width, canvas.height);
    
    if (totalWeight === 0) return;

    let startAngle = -Math.PI / 2; // 真上からスタート

    participants.forEach((p, index) => {
        const sliceAngle = (p.weight / totalWeight) * 2 * Math.PI;
        
        ctx.beginPath();
        ctx.moveTo(150, 150);
        ctx.arc(150, 150, 150, startAngle, startAngle + sliceAngle);
        ctx.closePath();
        
        ctx.fillStyle = colors[index % colors.length];
        ctx.fill();
        ctx.stroke();

        // 文字の描画
        ctx.save();
        ctx.translate(150, 150);
        ctx.rotate(startAngle + sliceAngle / 2);
        ctx.textAlign = "right";
        ctx.fillStyle = "#333";
        ctx.font = "16px sans-serif";
        ctx.fillText(p.name, 130, 5);
        ctx.restore();

        startAngle += sliceAngle;
    });
}

function spin() {
    if (participants.length === 0) return;

    const canvas = document.getElementById('rouletteCanvas');
    const totalWeight = participants.reduce((sum, p) => sum + p.weight, 0);
    
    // 重み付けによるランダム抽選
    let random = Math.random() * totalWeight;
    let winnerIndex = 0;
    for (let i = 0; i < participants.length; i++) {
        if (random < participants[i].weight) {
            winnerIndex = i;
            break;
        }
        random -= participants[i].weight;
    }

    // 当選者が真上にくるような角度を計算し、ランダムな回転（最低5周）を追加
    const winner = participants[winnerIndex];
    let startAngle = 0;
    for (let i = 0; i < winnerIndex; i++) {
        startAngle += (participants[i].weight / totalWeight) * 360;
    }
    const winnerAngle = (winner.weight / totalWeight) * 360;
    
    // 真上(270度)に持ってくるためのターゲット角度計算
    const targetDegree = 360 - (startAngle + winnerAngle / 2);
    
    // 最低5周（1800度）回す
    const spins = 360 * 5;
    currentRotation += spins + targetDegree - (currentRotation % 360);

    // CSSで回転アニメーションを実行
    canvas.style.transform = `rotate(${currentRotation}deg)`;

    // アニメーション終了(3秒後)に結果を表示
    setTimeout(() => {
        document.getElementById('result').textContent = `決定: ${winner.name}！`;
    }, 3000);
}

Dockerfile
超軽量なNginxイメージを使用して、これらの静的ファイルをホスティングします。
# 軽量なNginxのAlpineベースイメージを使用
FROM nginx:alpine

# ローカルのファイルをコンテナ内のNginx公開ディレクトリにコピー
COPY index.html /usr/share/nginx/html/
COPY style.css /usr/share/nginx/html/
COPY script.js /usr/share/nginx/html/

# ポート80を開放
EXPOSE 80

3. コンテナの起動方法
Dockerがインストールされている環境で、ファイルを作成したディレクトリ（roulette-app）に移動し、以下のコマンドを実行します。
1. イメージのビルド
docker build -t roulette-app .

2. コンテナの起動
docker run -d -p 8080:80 --name my-roulette roulette-app

これで、ブラウザから http://localhost:8080 にアクセスすれば、ルーレットアプリが利用できます！

    const weight = parseInt(document.getElementById('weightInput').value);

    if (name && weight > 0) {
        participants.push({ name, weight });
        document.getElementById('nameInput').value = '';
        document.getElementById('weightInput').value = '1'; // 追加後はデフォルトの1に戻す
        updateList();
        drawRoulette();
    }

// ★ここを改修：リストの描画と、変更・削除機能の追加
function updateList() {
    const list = document.getElementById('participantList');
    list.innerHTML = '';
    
    participants.forEach((p, index) => {
        const li = document.createElement('li');
        
        // 名前
        const nameSpan = document.createElement('span');
        nameSpan.textContent = p.name;
        
        // 比率変更用の入力欄
        const weightInput = document.createElement('input');
        weightInput.type = 'number';
        weightInput.value = p.weight;
        weightInput.min = '1';
        weightInput.className = 'edit-weight';
        // 値が変わった瞬間に比率を更新してルーレットを再描画
        weightInput.onchange = (e) => updateWeight(index, e.target.value);
        
        // 削除ボタン
        const deleteBtn = document.createElement('button');
        deleteBtn.textContent = '×';
        deleteBtn.className = 'delete-btn';
        deleteBtn.onclick = () => removeParticipant(index);

        li.appendChild(nameSpan);
        li.appendChild(weightInput);
        li.appendChild(deleteBtn);
        list.appendChild(li);
    });
}

// 比率の更新処理
function updateWeight(index, newWeight) {
    const weight = parseInt(newWeight);
    if (weight > 0) {
        participants[index].weight = weight;
        drawRoulette(); // 数値が変わったら即座にグラフに反映
    }
}

// 参加者の削除処理
function removeParticipant(index) {
    participants.splice(index, 1);
    updateList();
    drawRoulette();
}
